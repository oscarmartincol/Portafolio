        Escribir "¿Cuál es su nombre?";

    Java

     
    import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        Persona persona = new Persona();
        // imprimir mensaje solicitando el nombre de la persona.
        System.out.println("Ingrese su(s) nombre(s):");
        //Linea de código para asignar información al atributo nombre del objeto persona.
        persona.setNombre(input.nextLine());
        // Imprimir mensaje solicitando el apellido de la persona.
        System.out.println("Ingrese su(s) apellido(s):");
        //Linea de código para asignar información al atributo apellido del objeto persona.
        persona.setApellidos(input.nextLine());
        // Imprimir mensaje solicitando la edad de la persona.
        System.out.println("Ingrese su edad:");
        //Linea de código para asignar información al atributo edad del objeto persona.
        persona.setEdad(input.nextInt());
        // Imprimir mensaje solicitando la estatura de la persona.
        System.out.println("Ingrese su estatura:");
        //Linea de código para asignar información al atributo estatura del objeto persona.
        // Por teclado el punto decimal se debe escribir como ,
        persona.setEstatura(input.nextFloat());
        //Linea de código para imprimir los nombres, apellidos, edad y estatura de la persona.
        System.out.println(persona.toString());

    }
}

public class Persona {
    private String nombre;
    private int edad;
    private String apellidos;
    private double estatura;

    public void setNombre(String nombre) {
        this.nombre = nombre;
    }

    public void setEdad(int edad) {
        this.edad = edad;
    }

    public void setApellidos(String apellidos) {
        this.apellidos = apellidos;
    }

    public void setEstatura(double estatura) {
        this.estatura = estatura;
    }

    @Override
    public String toString() {
        return  "nombre='" + nombre + '\'' +
                ", edad=" + edad +
                ", apellidos='" + apellidos + '\'' +
                ", estatura=" + estatura;
    }
}

    Códigos de la solución

    Pseint

     
        Proceso punto_3
        Definir nombre Como Caracter;
        Definir apellidos Como Caracter;
        Definir nombreMadre Como Caracter;
        Definir nombrePadre Como Caracter;
                            
        Escribir "Cuál es su nombre";

        Escribir "Nombre completo de Madre: ";
        Leer nombreMadre;
        Escribir "Nombre completo del padre: ";
        Leer nombrePadre;

    Java

     
    import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        Persona persona = new Persona();
        Persona madre = new Persona();
        Persona padre = new Persona();
        // imprimir mensaje solicitando el nombre de la persona.
        System.out.println("Ingrese su(s) nombre(s):");
        //Linea de código para asignar información al atributo nombre del objeto persona.
        persona.setNombre(input.nextLine());
        // imprimir mensaje solicitando el apellido de la persona.
        System.out.println("Ingrese su(s) apellido(s):");
        //Linea de código para asignar información al atributo apellido del objeto persona.
        persona.setApellido(input.nextLine());
        // imprimir mensaje solicitando el nombre de la madre.
        System.out.println("Ingrese el nombre de tu madre:");
        //Linea de código para asignar información al atributo nombre del objeto madre.
        madre.setNombre(input.nextLine());
        // imprimir mensaje solicitando el apellido de la madre.
        System.out.println("Ingrese los apellidos de tu madre:");
        //Linea de código para asignar información al atributo apellido del objeto persona.
        madre.setApellido(input.nextLine());
        // imprimir mensaje solicitando el nombre del padre.
        System.out.println("Ingrese el nombre de tu padre:");
        //Linea de código para asignar información al atributo nombre del objeto padre.
        padre.setNombre(input.nextLine());
        // imprimir mensaje solicitando el apellido de la padre.
        System.out.println("Ingrese los apellidos de tu padre:");
        //Linea de código para asignar información al atributo apellido del objeto padre.
        padre.setApellido(input.nextLine());
        System.out.println(imprimirMensaje(persona.getNombre(), persona.getApellido(), madre.getNombre(),
                madre.getApellido(), padre.getNombre(), padre.getApellido()));
    }
    //Método para imprimir el mensaje solicitado.
    public static String imprimirMensaje(String nombre,String apellido, String nombreMadre, String
            apellidosMadre, String nombrePadre, String apellidosPadre){
        return "Yo " + nombre + ' ' + apellido + ", soy hijo de " + nombreMadre + ' ' + apellidosMadre
                + " y " + nombrePadre + ' ' + apellidosPadre;
    }
}

  4. Realizar un programa el cual solicite el nombre de una ciudad capital y el país. Al finalizar debe imprimir en pantalla el siguiente mensaje "La ciudad [Nombre de la Ciudad], es la capital del país [Nombre del País]

    Códigos de la solución

    Pseint

     
        Proceso Punto_4
        Definir ciudad Como Caracter;
        Definir pais Como Caracter;
                            
        Escribir "¿Nombre de la ciudad?";
        Leer ciudad;
        Escribir "¿De que país?";
        Leer pais;
                            
        Escribir "La ciudad ", ciudad, ", es la capital del país ", pais;
    FinProceso

    Java

     
    import java.util.Scanner;

public class Main {
    // Creación de variables.
    private static String ciudad;
    private static String pais;
    public static void main(String[] args){
        Scanner input = new Scanner(System.in);
    // Solicitar información de ciudad y pais
        System.out.print("Ingrese la ciudad: ");
        ciudad = input.nextLine();
        System.out.print("Ingrese el pais: ");
        pais = input.nextLine();
        System.out.println(imprimirFrase(ciudad, pais));
    }
    private static String imprimirFrase(String ciudad, String pais){
        return "La ciudad " + ciudad + ", es la capital del país " + pais;
    }
}
  5. Realizar un programa el cual solicite el nombre de su mascota, edad de la mascota, el tipo de mascota y su nombre completo. Al finalizar el sistema debe e imprimir en pantalla el siguiente mensaje: [Nombre de Mascota] es un(a) [Tipo de Mascota], el cual, tiene [Edad de la Mascota] años de edad y [Nombre Completo] es actualmente su dueño(a).

    Códigos de la solución

    Pseint

     
        Proceso Punto_5
        Definir nombreMascota Como Caracter;
        Definir edadMascota Como Entero;
        Definir tipoMascota Como Caracter;
        Definir nombreCompleto Como Caracter;
        
        Escribir "¿Cuál es el nombre de la mascota?";
        Leer nombreMascota;
        Escribir "¿Cuál es la edad?";
        Leer edadMascota;
        Escribir "¿Qué tipo de mascota es?";
        Leer tipoMascota;
        Escribir "¿Cuál es el nombre del dueño?";
        Leer nombreCompleto;
        
        Escribir nombreMascota," es un(a) ",tipoMascota, ", el cual, tiene ", edadMascota,

    FinProceso

    Java

     
    import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        Mascota mascota = new Mascota();
        // Solicitud de datos de la mascota
        System.out.print("Escriba el nombre de la mascota: ");
        mascota.setNombre(input.nextLine());
        System.out.print("Escriba el tipo de mascota: ");
        mascota.setTipo(input.nextLine());
        System.out.print("Nombre completo del dueño: ");
        mascota.setDueno(input.nextLine());
        System.out.print("Edad de la mascota: ");
        mascota.setEdad(input.nextInt());
        // imprimir mensaje solicitado
        System.out.println(imprimirMensaje(mascota.getNombre(), mascota.getTipo(),
                mascota.getDueno(), mascota.getEdad()));
    }
    public static String imprimirMensaje(String nombre, String tipo, String dueno, int edad){
        return nombre + " es un(a) " + tipo + ", el cual, tiene "
                + edad + " años de edad y " + dueno + " es actualmente su dueño(a).";
    }
}

public class Mascota {

        this.nombre = nombre;

    public void setTipo(String tipo) {
        this.tipo = tipo;
    }

    public int getEdad() {
        return edad;
    }

    public void setEdad(int edad) {
        this.edad = edad;
    }

        return dueno;
    }

    public void setDueno(String dueno) {
        this.dueno = dueno;
    }
}
