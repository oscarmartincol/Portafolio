  1. Realizar un programa el cual solicite su nombre y apellidos.

    Códigos de la solución

        Definir nombre Como Caracter;
        Definir apellidos Como Caracter;
                            
        Escribir "¿Cuál es su nombre?";

        Escribir "¿Cuales son tus apellidos?";

    FinProceso

    Java

     
                                import java.util.Scanner;
    public class Main {
        public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        Persona persona = new Persona();
        // imprimir mensaje solicitando el nombre de la persona.
        System.out.println("Ingrese su(s) nombre(s):");
        //Linea de código para asignar información al atributo nombre del objeto persona.
        persona.setNombre(input.nextLine());
        // imprimir mensaje solicitando el apellido de la persona.
        System.out.println("Ingrese su(s) apellido(s):");
        //Linea de código para asignar información al atributo apellido del objeto persona.
        persona.setApellido(input.nextLine());
        //Linea de código para imprimir los nombres y apellidos de la persona.
        System.out.println(persona.toString());
    }
    }

    public class Persona {
        private String nombre;
        private String apellido;
    
        public String getNombre() {
            return nombre;
        }
    
        public String getApellido() {
            return apellido;
        }
    
        public void setNombre(String nombre) {
            this.nombre = nombre;
        }
    
        public void setApellido(String apellido) {
            this.apellido = apellido;
        }
    
        @Override
        public String toString() {
            return  "nombres ='" + nombre + '\'' +
                    ", apellidos='" + apellido + '\'';
        }
    }
